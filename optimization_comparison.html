#include <stdio.h>
int GLOBAL_FEE = 15;

int apply_tax(int amount) {
    return amount + 5;            // Simple function — inlining candidate
}

int process_batch(int count, int rate) {
    return (count * rate) + (count * rate);  // CSE: (count*rate) computed twice
}

int main() {
    long long total_balance = 0;
    int base_val = 25;            // Constant propagation target

    for (int i = 0; i < 1000000; i++) {
        int step1 = base_val + 15; // Constant folding → 40
        int step2 = step1;           // Copy propagation chain
        int final_step = step2;      // Copy propagation chain

        int scaled_idx = i * 4;     // Strength reduction → i << 2
        int normalized = (scaled_idx * 1) + 0; // Algebraic simplification

        int p = 12, q = 8;
        int check1 = p * q;          // CSE: same expression
        int check2 = p * q;          // CSE: eliminated

        int temp_audit = 500;       // Dead code — never used

        if (i < -1) {               // Unreachable: i starts at 0
            total_balance += 999;    // Dead code — never executed
        }

        int static_bias = 40 * 2;  // Constant folding → 80
        total_balance += apply_tax(33) + final_step + static_bias;
        total_balance += GLOBAL_FEE; // Loop-invariant (hoistable)
    }

    for (int j = 0; j < 1000000; j++) {
        total_balance += process_batch(10, 5); // Constant args → inlined
    }

    printf("Final Balance: %lld\n", total_balance);
    return 0;
}
